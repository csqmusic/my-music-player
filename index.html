    // 辅助函数: 时间格式化 mm:ss

    function formatTime(seconds) {

        if (isNaN(seconds)) return "0:00";

        const mins = Math.floor(seconds / 60);

        const secs = Math.floor(seconds % 60);

        return `${mins}:${secs < 10 ? '0' + secs : secs}`;

    }


    // 解析LRC文本 -> 返回数组 [{time, text}]

    function parseLRC(lrcText) {

        const lines = lrcText.split('\n');

        const parsed = [];

        const timeRegex = /\[(\d{2}):(\d{2})\.(\d{2,3})\]/;

        for (let line of lines) {

            line = line.trim();

            if (!line) continue;

            const match = line.match(timeRegex);

            if (match) {

                const minutes = parseInt(match[1], 10);

                const seconds = parseInt(match[2], 10);

                const millis = parseInt(match[3], 10);

                let timeSec = minutes * 60 + seconds + millis / 1000;

                const text = line.replace(timeRegex, '').trim();

                if (text) {

                    parsed.push({ time: timeSec, text: text });

                }

            } else {

                // 没有时间标签的文本忽略或者作为默认？忽略

            }

        }

        // 按时间排序

        parsed.sort((a,b) => a.time - b.time);

        return parsed;

    }


    // 加载歌曲: 切换src, 更新UI, 解析歌词

    function loadSong(index) {

        if (index < 0) index = 0;

        if (index >= SONGS.length) index = 0;

        currentSongIndex = index;

        const song = SONGS[currentSongIndex];

        // 更新UI文本

        currentSongTitleSpan.innerText = song.name;

        currentArtistSpan.innerText = song.artist;

        // 切换音频源

        const wasPlaying = isPlaying;

        audio.pause();

        audio.src = song.src;

        audio.load();

        // 解析歌词

        currentLyrics = parseLRC(song.lrc);

        renderLyrics(0); // 重置歌词显示，高亮第0行或占位

        // 重置进度条

        progressFilled.style.width = '0%';

        currentTimeSpan.innerText = '0:00';

        durationSpan.innerText = '0:00';

        

        // 如果加载前正在播放，则自动播放 (由于load后需要等待canplay, 但为了手势限制, 调用play)

        if (wasPlaying) {

            const playPromise = audio.play();

            if (playPromise !== undefined) {

                playPromise.then(() => {

                    isPlaying = true;

                    updatePlayButtonIcon(true);

                }).catch(() => {

                    // 自动播放被阻止，暂停状态

                    isPlaying = false;

                    updatePlayButtonIcon(false);

                });

            }

        } else {

            isPlaying = false;

            updatePlayButtonIcon(false);

        }

        

        // 更新歌单高亮

        highlightActiveSong();

        // 获取总时长后更新

        audio.onloadedmetadata = () => {

            if (!isNaN(audio.duration)) {

                durationSpan.innerText = formatTime(audio.duration);

            }

        };

    }


    // 根据当前播放时间更新歌词高亮

    function updateLyricsByTime(currentTime) {

        if (!currentLyrics.length) {

            if (lyricsContentDiv.innerHTML.trim() === '' || lyricsContentDiv.children.length === 0) {

                lyricsContentDiv.innerHTML = '<div class="lyric-line">🎵 暂无滚动歌词，享受旋律吧 🎵</div>';

            }

            return;

        }

        let activeIndex = -1;

        for (let i = 0; i < currentLyrics.length; i++) {

            if (currentTime >= currentLyrics[i].time) {

                activeIndex = i;

            } else {

                break;

            }

        }

        if (activeIndex === -1 && currentLyrics.length > 0 && currentTime < currentLyrics[0].time) {

            activeIndex = 0;

        }

        renderLyrics(activeIndex);

    }

    

    // 渲染歌词面板，高亮某一行并自动滚动

    function renderLyrics(activeIndex) {

        if (!currentLyrics.length) {

            lyricsContentDiv.innerHTML = '<div class="lyric-line">✨ 纯音乐 / 无歌词 ✨</div>';

            return;

        }

        let html = '';

        for (let i = 0; i < currentLyrics.length; i++) {

            const line = currentLyrics[i].text;

            const activeClass = (i === activeIndex) ? 'active' : '';

            html += `<div class="lyric-line ${activeClass}" data-line-index="${i}">${escapeHtml(line)}</div>`;

        }

        lyricsContentDiv.innerHTML = html;

        if (activeIndex >= 0) {

            const activeElement = lyricsContentDiv.querySelector(`.lyric-line.active`);

            if (activeElement) {

                activeElement.scrollIntoView({ behavior: 'smooth', block: 'center' });

            }

        }

    }

    

    // 简单防XSS

    function escapeHtml(str) {

        return str.replace(/[&<>]/g, function(m) {

            if (m === '&') return '&amp;';

            if (m === '<') return '&lt;';

            if (m === '>') return '&gt;';

            return m;

        });

    }


    // 播放/暂停切换

    function togglePlayPause() {

        if (audio.src === '') {

            loadSong(currentSongIndex);

        }

        if (isPlaying) {

            audio.pause();

            isPlaying = false;

            updatePlayButtonIcon(false);

        } else {

            const playPromise = audio.play();

            if (playPromise !== undefined) {

                playPromise.then(() => {

                    isPlaying = true;

                    updatePlayButtonIcon(true);

                }).catch(err => {

                    console.warn("播放失败:", err);

                    isPlaying = false;

                    updatePlayButtonIcon(false);

                });

            }

        }

    }

            isPlaying = false;

            updatePlayButtonIcon(false);

    // 进度条拖动

    function setProgress(e) {

        const rect = progressBar.getBoundingClientRect();

        let clickX = e.clientX - rect.left;

        let percent = clickX / rect.width;

        percent = Math.min(1, Math.max(0, percent));

        if (audio.duration && !isNaN(audio.duration)) {

            audio.currentTime = percent * audio.duration;

        }

    }

    

    // 音量控制

    function setVolume(e) {

        const rect = volumeBar.getBoundingClientRect();

        let clickX = e.clientX - rect.left;

        let vol = clickX / rect.width;

        vol = Math.min(1, Math.max(0, vol));

        audio.volume = vol;

        volumeFill.style.width = (vol * 100) + '%';

    }

        });

        audio.addEventListener('canplay', () => {

            if (!isNaN(audio.duration)) durationSpan.innerText = formatTime(audio.duration);

        });

    }

    function login(username, password) {

            if (currentLyrics.length === 0 && SONGS[currentSongIndex]) {

                currentLyrics = initLrc;
